DROP TABLE DEPARTMENT CASCADE CONSTRAINTS;
DROP TABLE REPRESENTATIVES CASCADE CONSTRAINTS;
DROP TABLE CUSTOMER CASCADE CONSTRAINTS;
DROP TABLE INVOICE CASCADE CONSTRAINTS;
DROP TABLE INVOICE_LINE CASCADE CONSTRAINTS;



CREATE TABLE DEPARTMENT
	(department_code NUMBER(6),
	 department_name VARCHAR2(100) NOT NULL,
     department_location VARCHAR2(200) NOT NULL,
	 CONSTRAINT department_code_id PRIMARY KEY (department_code));

CREATE TABLE REPRESENTATIVES
	(rep_code NUMBER(6),
	 department_code NUMBER(6) NOT NULL,
	 rep_lname VARCHAR2(50)NOT NULL,
	 CONSTRAINT rep_code_id PRIMARY KEY(rep_code),
	 CONSTRAINT rep_department_code_fk FOREIGN KEY (department_code) REFERENCES DEPARTMENT(department_code));

CREATE TABLE CUSTOMER
	(cust_code NUMBER(6),
	 cust_name VARCHAR2(50) NOT NULL,
	 cust_address VARCHAR2(100) NOT NULL,
	 CONSTRAINT cust_code_id PRIMARY KEY (cust_code));

CREATE TABLE INVOICE
	(inv_num NUMBER(6),
	 rep_code NUMBER(6) NOT NULL,
	 cust_code NUMBER(6) NOT NULL,
	 inv_amt NUMBER(13,2) NOT NULL,
	 CONSTRAINT inv_num_pk PRIMARY KEY (inv_num),
	 CONSTRAINT invoice_rep_code_id FOREIGN KEY(rep_code) REFERENCES REPRESENTATIVES(rep_code),
	 CONSTRAINT invoice_cust_code_id FOREIGN KEY (cust_code) REFERENCES CUSTOMER(cust_code));
	 
CREATE TABLE INVOICE_LINE
	(inv_num NUMBER(6),
	 inv_line_num NUMBER(6),
	 item_code VARCHAR2(20) NOT NULL,
	 CONSTRAINT inv_num_fk_id FOREIGN KEY(inv_num) REFERENCES INVOICE(inv_num),
	 CONSTRAINT invoice_num_id_line_id PRIMARY KEY (inv_num, inv_line_num));
	 

DESCRIBE DEPARTMENT;
DESCRIBE REPRESENTATIVES;
DESCRIBE CUSTOMER;
DESCRIBE INVOICE;
DESCRIBE INVOICE_LINE;
