CREATE TABLE DEPARTMENT
	(DEPARTMENT_CODE NUMBER(6),
	 DEPARTMENT_NAME VARCHAR2(100),
         DEPARTMENT_LOCATION VARCHAR2(200),
	 CONSTRAINT DEPARTMENT_CODE_ID PRIMARY KEY);

CREATE TABLE REPRESENTATIVES
	(REP_CODE NUMBER(6),
	 REP_LNAME VARCHAR2(50),
	 CONSTRAINT REP_CODE_ID PRIMARY KEY,
	 CONSTRAINT REP_CODE_ID_FK
	 FOREIGN KEY (DEPARTMENT_CODE)
	 REFERENCES DEPARTMENT (DEPARTMENT_CODE));

CREATE TABLE CUSTOMER
	(CUST_CODE NUMBER(6),
	 CUST_NAME VARCHAR2(50),
	 CUST_ADDRESS VARCHAR2(100),
	 CONSTRAINT CUST_CODE_ID PRIMARY KEY);
